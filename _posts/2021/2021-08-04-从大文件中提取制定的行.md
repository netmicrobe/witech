~~~bash
# using sed:

# 10 - 20 行
sed -n '10,20p' in.sql > out.sql
# 第一行，加上，10 - 20 行
sed -n '1p; 10,20p' in.sql > out.sql


# using awk:
awk 'NR>=16224&&NR<=16482' in.sql > out.sql
~~~
